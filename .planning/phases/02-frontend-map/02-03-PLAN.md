```tsx
import { useMemo } from 'react';
import { Marker, Tooltip } from 'react-leaflet';
import L from 'leaflet';
import type { CityTemperature } from '../types/cityTemperature';

// NOTE: 'leaflet/dist/leaflet.css' is imported in BulgariaMap.tsx — do NOT import it here

function getTempColorClass(temp: number): string {
  if (temp <= 0)  return 'bg-blue-500';   // ≤0°C  (D-09)
  if (temp <= 15) return 'bg-teal-500';   // 1–15°C (D-09)
  if (temp <= 25) return 'bg-green-500';  // 16–25°C (D-09)
  if (temp <= 35) return 'bg-orange-500'; // 26–35°C (D-09)
  return 'bg-red-500';                    // >35°C  (D-09)
}

function createTempIcon(temp: number): L.DivIcon {
  const colorClass = getTempColorClass(temp);
  const label = isNaN(temp) ? '—' : `${temp}°C`;  // null/NaN guard (UI-SPEC data contract)
  return L.divIcon({
    className: '',  // D-07: CRITICAL — clears leaflet-div-icon default white box so Tailwind controls all styling
    html: `<div class="${colorClass} text-white text-[13px] font-bold px-2 py-1 rounded shadow-md whitespace-nowrap leading-none select-none">${label}</div>`,
    iconSize:   [52, 28],  // UI-SPEC §2: bounding box for hit-test and centering
    iconAnchor: [26, 14], // UI-SPEC §2: center of badge = exact city coordinate (half of iconSize)
  });
}

export function CityMarker({ city }: { city: CityTemperature }) {
  const roundedTemp = Math.round(city.temperatureC ?? NaN);  // D-08: integer display; ?? NaN handles null/undefined

  // PITFALL §6: icon MUST be memoized — never create L.divIcon inline in JSX
  // Recreated only when temperature changes, not on every parent re-render
  const icon = useMemo(
    () => createTempIcon(roundedTemp),
    [roundedTemp]
  );

  return (
    // PITFALL §13: use city.latitude/city.longitude (canonical backend coords), not Open-Meteo snapped coords
    <Marker position={[city.latitude, city.longitude]} icon={icon}>
      <Tooltip direction="top" permanent={false} opacity={0.9}>
        {city.nameEn}
      </Tooltip>
    </Marker>
  );
}
```

Critical implementation points:
- **`className: ''` is non-negotiable (D-07)**: Without it, Leaflet adds `leaflet-div-icon` CSS which creates a white box with border that overrides Tailwind classes — the badge will look broken.
- **`useMemo([roundedTemp])` is non-negotiable (PITFALL §6)**: Creating the DivIcon inline in JSX causes all 28 markers to flicker/remount on every React state change. The memo dependency array must be `[roundedTemp]`, not `[city.temperatureC]` (using the rounded value avoids unnecessary recreations for sub-integer temperature fluctuations).
- **`city.temperatureC ?? NaN`**: The `??` operator handles `null` and `undefined` (not `0`, which is a valid temperature). Then `Math.round(NaN)` → `NaN`, and `isNaN(NaN)` → `true` → renders `—`.
- **Badge HTML classes** (UI-SPEC §2 — exact classes, do not modify): `{colorClass} text-white text-[13px] font-bold px-2 py-1 rounded shadow-md whitespace-nowrap leading-none select-none`
- **No leaflet.css import here**: BulgariaMap.tsx imports it — double-importing causes no error but wastes a network request.

```tsx
import { MapContainer, TileLayer } from 'react-leaflet';
import L from 'leaflet';
import 'leaflet/dist/leaflet.css';
import type { CityTemperature } from '../types/cityTemperature';
import { CityMarker } from './CityMarker';

// Leaflet default marker icon fix for bundlers (Vite/webpack asset hashing breaks default icon URLs)
// Harmless when using DivIcon-only markers, but prevents broken image errors if a fallback Marker is ever introduced
import markerIcon2x from 'leaflet/dist/images/marker-icon-2x.png';
import markerIcon from 'leaflet/dist/images/marker-icon.png';
import markerShadow from 'leaflet/dist/images/marker-shadow.png';

delete (L.Icon.Default.prototype as unknown as Record<string, unknown>)._getIconUrl;
L.Icon.Default.mergeOptions({
  iconUrl: markerIcon,
  iconRetinaUrl: markerIcon2x,
  shadowUrl: markerShadow,
});

// D-05: immutable after mount — set correctly from the start (PITFALL §7)
const BULGARIA_CENTER: [number, number] = [42.7, 25.5];

interface BulgariaMapProps {
  cities: CityTemperature[] | null;
}

export function BulgariaMap({ cities }: BulgariaMapProps) {
  return (
    // D-03: full-viewport wrapper; aria-label identifies the map for screen readers (UI-SPEC accessibility)
    <div
      className="h-screen w-screen"
      aria-label="Bulgaria weather map showing current temperatures for 28 district cities"
    >
      {/* D-05 / PITFALL §7: center and zoom are immutable — never pass them as dynamic state */}
      <MapContainer
        center={BULGARIA_CENTER}
        zoom={7}
        scrollWheelZoom
        className="h-full w-full"
      >
        {/* D-06: CartoDB Positron tiles — clean minimal basemap */}
        <TileLayer
          attribution='&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors &copy; <a href="https://carto.com/">CARTO</a>'
          url="https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}{r}.png"
        />
        {/* MAP-02: render one CityMarker per city; key on nameEn (unique English city names) */}
        {cities?.map((city) => (
          <CityMarker key={city.nameEn} city={city} />
        ))}
      </MapContainer>
    </div>
  );
}
```

Critical implementation points:
- **`center={BULGARIA_CENTER}` and `zoom={7}` MUST be set from the start as constants, never from state (PITFALL §7)**: react-leaflet's MapContainer ignores changes to these props after the initial render. If set incorrectly, the only fix is unmounting and remounting the component.
- **`className="h-full w-full"` on MapContainer**: The containing `<div>` has `h-screen w-screen`; MapContainer must fill it with `h-full w-full`. Without this, Leaflet renders a 0-height container and nothing is visible.
- **Leaflet attribution is left at default bottom-right position**: Do NOT remove attribution — it is required by CartoDB/OpenStreetMap tile license.
- **`cities?.map(...)` with optional chaining**: When `cities` is `null` (loading state), no markers are rendered — the map tiles still display correctly, with the loading overlay from App.tsx on top.
- **`key={city.nameEn}`**: English city names are unique identifiers. Do not use `city.nameNative` as key (Bulgarian characters, slightly riskier for React reconciliation edge cases).

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Backend API response → L.divIcon html string | `temperatureC` from JSON enters the badge HTML; must not allow arbitrary HTML injection |
| Backend API response → Leaflet Tooltip | `nameEn` from JSON rendered as Tooltip children; React renders this as escaped text |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | XSS | `createTempIcon` — `${label}` in `html` string | mitigate | `label` is derived exclusively from `Math.round(city.temperatureC ?? NaN)` which coerces to a number or NaN, plus `isNaN()` guard which replaces NaN with `'—'`. Attacker cannot inject arbitrary HTML via a numeric field. `getTempColorClass()` returns only 5 hardcoded Tailwind class strings based on numeric comparison — no user input touches the class name. Net result: no code path allows HTML injection into the divIcon template. |
| T-03-02 | XSS | Leaflet `<Tooltip>` children `{city.nameEn}` | accept | React renders children as escaped text nodes, not innerHTML. Even if `nameEn` contains `<script>` or `<img onerror=...>`, React escapes it to `&lt;script&gt;`. No sanitization needed. |
| T-03-03 | Spoofing | CartoDB tile URL — attacker could theoretically intercept tile requests | accept | Tiles are static map imagery with no PII or authentication. MITM on tile traffic reveals nothing sensitive. Acceptable for a development weather tool. |
| T-03-04 | Denial of Service | All 28 markers recreated if parent re-renders (without memoization) | mitigate | `useMemo([roundedTemp])` in CityMarker prevents icon recreation on unrelated parent state changes. PITFALL §6 explicitly calls this out. |
</threat_model>
